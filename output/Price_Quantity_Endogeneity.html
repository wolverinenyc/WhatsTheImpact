import pandas as pd
import numpy as np
import statsmodels.formula.api as smf
import matplotlib.pyplot as plt

We generate our own data so we know the precise relationships between features. Here, we have a set of 1000 products that have a distribution of 'quality.' The price is set proportional to the quality with some random noise: higher quality products have higher prices. The quantity sold increases with quality and decreases with price: all else equal, customers prefer higher quality and lower prices.¶

In [40]:
np.random.seed(123)
numproducts = 1000

quality = np.random.normal(loc=100, scale=25,size=numproducts)
price = np.sqrt(quality) + np.random.normal(loc=10, scale=2, size=numproducts)

log_qty_sold = np.maximum(0, (2*np.log(quality) -0.5*np.log(price)))
qty_sold = np.exp(log_qty_sold).round()

df = pd.DataFrame({'quality':quality, 'price':price, 'qty_sold':qty_sold})
