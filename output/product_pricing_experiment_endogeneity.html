np.random.seed(123)
numproducts = 1000

# products have a normal distribution of quality
quality = np.random.normal(loc=100, scale=25,size=numproducts)

# price is a positive function of quality
price = np.sqrt(quality) + np.random.normal(loc=10, scale=2, size=numproducts)

# quantity sold and price often have a log-log relationship so that a percentage increase in price 
# drives a percentage increase in quantity sold

log_price = np.log(price)
log_quality = np.log(quality)

# with this formula, we know the exact relationship between price, quality, and sales
log_qty_sold = np.maximum(0, (2*log_quality -0.5*log_price))

# typically, your data set will only include price and quantity, and you'll have to create the log transformations
qty_sold = np.exp(log_qty_sold).round()
df = pd.DataFrame({'quality':quality, 'price':price, 'qty_sold':qty_sold, 'log_quality':log_quality,
                  'log_price':log_price, 'log_qty_sold':log_qty_sold})
